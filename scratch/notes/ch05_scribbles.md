```python
async def execute_plan(self, plan: ActionPlan) -> AsyncIterator[Response]:
    """Execute and monitor the action plan."""

    # Track progress
    progress = ProgressTracker(plan)

    # Execute steps
    async for step in plan.steps:
        # Tool execution
        result = await self.execute_tool(step.tool, step.parameters)

        # Update progress
        progress.update(step, result)

        # Adapt if needed
        if not result.success:
            new_step = await self.adaptation_manager.handle_failure(step, result)
            yield await self.execute_tool(new_step.tool, new_step.parameters)

        # Record outcomes
        await self.outcome_recorder.record(step, result)
```
